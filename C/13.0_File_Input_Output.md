#include <stdio.h>
#include <stdlib.h>

int main()
{
	int pr;
	FILE* file;
	char file_path[20];
	char text[20];

	// write
	file = fopen(file_path, "w");
	if (!file) exit(1);

	for (int i = 0; text[i] != '\0'; ++i)
		fputc(text[i], file);

	fclose(file);

	// read
	file = fopen(file_path, "r");
	if (!file) exit(1);

	while ((ch = fgetc(file)) != EOF)
		putchar(ch);

	fclose(file);

	return 0;
}
